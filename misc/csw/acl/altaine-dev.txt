 permit tcp host 119.47.119.240 eq 53 any established
 permit tcp host 119.47.119.241 eq 53 any established

 !outbound ports

 permit tcp any eq 25 any established

 ! REF: XBL-910021
 permit tcp host 89.238.173.131 any established
